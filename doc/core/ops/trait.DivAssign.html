The DivAssign trait is used to specify the functionality of /=.

Examples

A trivial implementation of DivAssign. When Foo /= Foo happens, it ends up calling div_assign, and therefore, main prints Dividing!.

#![feature(augmented_assignments)] #![feature(op_assign_traits)] use std::ops::DivAssign; #[derive(Copy, Clone)] struct Foo; impl DivAssign for Foo { fn div_assign(&mut self, _rhs: Foo) { println!("Dividing!"); } } fn main() { let mut foo = Foo; foo /= Foo; } 
#![feature(augmented_assignments)]
#![feature(op_assign_traits)]

use std::ops::DivAssign;

#[derive(Copy, Clone)]
struct Foo;

impl DivAssign for Foo {
    fn div_assign(&mut self, _rhs: Foo) {
        println!("Dividing!");
    }
}

fn main() {
    let mut foo = Foo;
    foo /= Foo;
}
